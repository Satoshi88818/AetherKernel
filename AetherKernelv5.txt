    def H(self, a):

    def growth_deriv(self, a):

    # ====================== COSMOLOGICAL POISSON ======================
    def _poisson_fft_3d(self, delta, a):

    # ====================== CIC (exact mass conservation) ======================
    def _deposit_cic_3d(self, x, y, z):

        pos = np.stack([x, y, z], axis=-1) / self.dx
        i0 = np.floor(pos).astype(int) % self.Ngrid
        f = pos - i0
        one_minus_f = 1 - f

        for dx_ in [0, 1]:
            for dy_ in [0, 1]:
                for dz_ in [0, 1]:
                    w = (one_minus_f[...,0] if dx_==0 else f[...,0]) * \
                        (one_minus_f[...,1] if dy_==0 else f[...,1]) * \
                        (one_minus_f[...,2] if dz_==0 else f[...,2])
                    idx = (i0[...,0] + dx_) % self.Ngrid, \
                          (i0[...,1] + dy_) % self.Ngrid, \
                          (i0[...,2] + dz_) % self.Ngrid

        delta = rho - 1.0
        assert abs(delta.mean()) < 1e-10, f"Mass violation: mean(δ) = {delta.mean():.2e}"
        return delta

    def _interpolate_cic_3d(self, field, x, y, z):

        i0 = np.floor(pos).astype(int) % self.Ngrid

        one_minus_f = 1 - f
        val = np.zeros_like(x)
        for dx_ in [0, 1]:

                          (i0[...,1] + dy_) % self.Ngrid, \
                          (i0[...,2] + dz_) % self.Ngrid
                    val += w * field[idx]
        return val

    # ====================== LEAPFROG KDK INTEGRATOR ======================
    def leapfrog_kdk(self, x, y, z, vx, vy, vz, delta, a_old, da):
        """Kick-Drift-Kick with canonical momentum v = a dx/dt"""
        a_new = a_old + da

        fx_grid = -np.gradient(phi, self.dx, axis=0)
        fy_grid = -np.gradient(phi, self.dx, axis=1)
        fz_grid = -np.gradient(phi, self.dx, axis=2)
        fx = self._interpolate_cic_3d(fx_grid, x, y, z)
        fy = self._interpolate_cic_3d(fy_grid, x, y, z)
        fz = self._interpolate_cic_3d(fz_grid, x, y, z)
        vx += (dt / 2.0) * fx
        vy += (dt / 2.0) * fy
        vz += (dt / 2.0) * fz
        vx -= (dt / 2.0) * H_mid * vx
        vy -= (dt / 2.0) * H_mid * vy
        vz -= (dt / 2.0) * H_mid * vz

        # Drift full step
        x = (x + vx * dt / a_mid) % self.Lbox
        y = (y + vy * dt / a_mid) % self.Lbox
        z = (z + vz * dt / a_mid) % self.Lbox

        new_delta = self._deposit_cic_3d(x, y, z)

        # Kick ½
        phi = self._poisson_fft_3d(new_delta, a_new)
        fx_grid = -np.gradient(phi, self.dx, axis=0)
        fy_grid = -np.gradient(phi, self.dx, axis=1)
        fz_grid = -np.gradient(phi, self.dx, axis=2)
        fx = self._interpolate_cic_3d(fx_grid, x, y, z)
        fy = self._interpolate_cic_3d(fy_grid, x, y, z)
        fz = self._interpolate_cic_3d(fz_grid, x, y, z)
        vx += (dt / 2.0) * fx
        vy += (dt / 2.0) * fy
        vz += (dt / 2.0) * fz
        vx -= (dt / 2.0) * H_mid * vx  # Approximate with H_mid for simplicity
        vy -= (dt / 2.0) * H_mid * vy
        vz -= (dt / 2.0) * H_mid * vz

        return x, y, z, vx, vy, vz, new_delta

    # ====================== POWER SPECTRUM ======================
    def compute_power_spectrum(self, delta, nbins=40):
        delta_k = fftn(delta)
        kfreq = fftfreq(self.Ngrid, d=self.dx) * 2 * np.pi
        kx, ky, kz = np.meshgrid(kfreq, kfreq, kfreq, indexing='ij')

        kbins = np.logspace(np.log10(0.01), np.log10(np.pi / self.dx), nbins)
        Pk = np.zeros(nbins-1)
        counts = np.zeros(nbins-1)
        for i in range(len(kbins)-1):
            mask = (k >= kbins[i]) & (k < kbins[i+1])
            if mask.sum() > 0:
                Pk[i] = pk[mask].mean()
                counts[i] = mask.sum()
        kcenters = (kbins[:-1] + kbins[1:]) / 2
        return kcenters[counts > 0], Pk[counts > 0]

    # ====================== 2LPT INITIAL CONDITIONS ======================
    def compute_2lpt(self, delta_lin, a_init):

        # Solve for phi1 with ∇²φ1 = -δ1
        prefactor = 1.5 * self.Omega_m / a_init
        phi1 = self._poisson_fft_3d(-delta1 / prefactor, a_init)

        kfreq = fftfreq(self.Ngrid, d=self.dx) * 2 * np.pi
        kx, ky, kz = np.meshgrid(kfreq, kfreq, kfreq, indexing='ij')

        phi1_k = fftn(phi1)

        dxx_k = -kx**2 * phi1_k
        dyy_k = -ky**2 * phi1_k
        dzz_k = -kz**2 * phi1_k
        dxy_k = -kx * ky * phi1_k
        dxz_k = -kx * kz * phi1_k
        dyz_k = -ky * kz * phi1_k

        dxx = np.real(ifftn(dxx_k))
        dyy = np.real(ifftn(dyy_k))
        dzz = np.real(ifftn(dzz_k))
        dxy = np.real(ifftn(dxy_k))
        dxz = np.real(ifftn(dxz_k))
        dyz = np.real(ifftn(dyz_k))

        source = dxx * dyy + dxx * dzz + dyy * dzz - dxy**2 - dxz**2 - dyz**2

        # Solve for phi2 with ∇²φ2 = source
        phi2 = self._poisson_fft_3d(source / prefactor, a_init)

        return phi1, phi2

    # ====================== HALO FINDER ======================
    def find_halos(self, x, y, z, b=0.2, min_size=20):
        positions = np.stack([x.ravel(), y.ravel(), z.ravel()], axis=1)
        Npart = len(positions)
        mean_sep = self.Lbox / self.Ngrid
        ll = b * mean_sep
        tree = KDTree(positions)
        pairs = tree.query_pairs(ll)
        G = nx.Graph()
        G.add_nodes_from(range(Npart))
        G.add_edges_from(pairs)
        components = list(nx.connected_components(G))
        halos = [comp for comp in components if len(comp) >= min_size]
        return halos

            a_tensor = np.full((coords.shape[0], 1), a)

            y_train.append(delta_history[i].flatten())

        X_train = torch.tensor(np.vstack(X_train), dtype=torch.float32)
        y_train = torch.tensor(np.concatenate(y_train), dtype=torch.float32).unsqueeze(1)

        optimizer = optim.Adam(model.parameters(), lr=lr)
        scheduler = optim.lr_scheduler.StepLR(optimizer, step_size=200, gamma=0.5)

        a_min = min(a_history)
        a_max = max(a_history)

        for ep in range(epochs):
            optimizer.zero_grad()
            pred = model(X_train)
            loss_data = nn.MSELoss()(pred, y_train)

            # Physics loss
            col_coords = torch.rand(n_col, 3)
            col_a = torch.rand(n_col, 1) * (a_max - a_min) + a_min
            col_inputs = torch.cat([col_coords, col_a], dim=1)
            col_inputs.requires_grad = True
            col_pred = model(col_inputs)
            partial_a = torch.autograd.grad(col_pred, col_inputs, grad_outputs=torch.ones_like(col_pred), create_graph=True)[0][:, -1].unsqueeze(1)
            f_a = torch.zeros_like(col_a)
            for i in range(n_col):
                aa = col_a[i].item()
                f_a[i] = self.growth_deriv(aa) / self.growth_factor(aa)
            res = partial_a - f_a * col_pred
            loss_phys = torch.mean(res**2)

            loss = loss_data + lambda_phys * loss_phys

            if ep % 100 == 0:
                loss += 0.01 * torch.mean(pred)**2
            loss.backward()
            optimizer.step()
            scheduler.step()
            if ep % 100 == 0:
                print(f"   Epoch {ep:4d} | Loss {loss.item():.2e} (data: {loss_data.item():.2e}, phys: {loss_phys.item():.2e})")

        self.pinn_model = model
        print("✅ PINN surrogate trained")
        return model

    def predict_pinn(self, a):
        if not hasattr(self, 'pinn_model'):
            raise RuntimeError("Train PINN first")
        x_grid = np.linspace(0, 1, self.Ngrid)
        X, Y, Z = np.meshgrid(x_grid, x_grid, x_grid, indexing='ij')

        inputs = torch.tensor(np.hstack([coords, np.full((len(coords), 1), a)]), dtype=torch.float32)
        with torch.no_grad():
            return self.pinn_model(inputs).numpy().reshape(self.Ngrid, self.Ngrid, self.Ngrid)
