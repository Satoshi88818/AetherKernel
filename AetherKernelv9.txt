# ====================== NUMBA-ACCELERATED CIC (unchanged – ultra-fast) ======================
@njit(fastmath=True, cache=True)
def _deposit_cic_numba(x, y, z, Lbox, Ngrid, dx):
    """Fast CIC deposition using np.add.at – ~10-20× faster than per-particle loop."""
    rho = np.zeros((Ngrid, Ngrid, Ngrid), dtype=np.float64)
    pos = np.stack((x, y, z), axis=-1) / dx
    i0 = np.floor(pos).astype(np.int32) % Ngrid
    f = pos - i0.astype(np.float64)
    one_minus_f = 1.0 - f
    Np = len(x)

    for dx_ in [0, 1]:
        for dy_ in [0, 1]:
            for dz_ in [0, 1]:
                w = (one_minus_f[:, 0] if dx_ == 0 else f[:, 0]) * \
                    (one_minus_f[:, 1] if dy_ == 0 else f[:, 1]) * \
                    (one_minus_f[:, 2] if dz_ == 0 else f[:, 2])
                idx0 = (i0[:, 0] + dx_) % Ngrid
                idx1 = (i0[:, 1] + dy_) % Ngrid
                idx2 = (i0[:, 2] + dz_) % Ngrid
                np.add.at(rho, (idx0, idx1, idx2), w)

    rho *= dx ** 3
    return rho


@njit(fastmath=True, cache=True)
def _interpolate_cic_numba(field, x, y, z, Lbox, Ngrid, dx):
    """Fast CIC interpolation – fully vectorized per octant."""
    val = np.zeros(len(x), dtype=np.float64)
    pos = np.stack((x, y, z), axis=-1) / dx
    i0 = np.floor(pos).astype(np.int32) % Ngrid
    f = pos - i0.astype(np.float64)
    one_minus_f = 1.0 - f

    for dx_ in [0, 1]:
        for dy_ in [0, 1]:
            for dz_ in [0, 1]:
                w = (one_minus_f[:, 0] if dx_ == 0 else f[:, 0]) * \
                    (one_minus_f[:, 1] if dy_ == 0 else f[:, 1]) * \
                    (one_minus_f[:, 2] if dz_ == 0 else f[:, 2])
                idx0 = (i0[:, 0] + dx_) % Ngrid
                idx1 = (i0[:, 1] + dy_) % Ngrid
                idx2 = (i0[:, 2] + dz_) % Ngrid
                field_vals = field[idx0, idx1, idx2]
                val += w * field_vals
    return val

            return k**2 / (2 * np.pi**2) * pk_func(k) * W**2
        sigma2 = quad(integrand, 1e-4, 10.0, epsrel=1e-4)[0]
        return np.sqrt(sigma2)

        fy = jnp.real(jnp.fft.ifftn(fy_k))

    def compute_power_spectrum(self, delta, nbins=40):

        k = np.sqrt(self.k2_mesh).flatten()
        pk = (np.abs(delta_k)**2 / self.Ngrid**6).flatten() * self.Lbox**3

        kbins = np.logspace(np.log10(0.01), np.log10(np.pi / self.dx), nbins)
        Pk = np.zeros(nbins-1)
        counts = np.zeros(nbins-1)
        for i in range(len(kbins)-1):
            mask = (k >= kbins[i]) & (k < kbins[i+1])
            if mask.sum() > 0:
                Pk[i] = pk[mask].mean()
                counts[i] = mask.sum()
        kcenters = (kbins[:-1] + kbins[1:]) / 2
        return kcenters[counts > 0], Pk[counts > 0]

    def compute_2lpt(self, delta_lin, a_init):

        dyy = np.real(ifftn(-self.ky**2 * phi1_k))
        dzz = np.real(ifftn(-self.kz**2 * phi1_k))
        dxy = np.real(ifftn(-self.kx * self.ky * phi1_k))
        dxz = np.real(ifftn(-self.kx * self.kz * phi1_k))
        dyz = np.real(ifftn(-self.ky * self.kz * phi1_k))

        source = dxx * dyy + dxx * dzz + dyy * dzz - dxy**2 - dxz**2 - dyz**2
        phi2 = self._poisson_fft_3d(source / prefactor, a_init)
        phi2_k = fftn(phi2)
        return phi1, phi2

    def find_halos(self, x, y, z, b=0.2, min_size=20):
        positions = np.stack([x.ravel(), y.ravel(), z.ravel()], axis=1)
        mean_sep = self.Lbox / self.Ngrid
        ll = b * mean_sep
        tree = KDTree(positions)
        pairs = tree.query_pairs(ll)
        G = nx.Graph()
        G.add_nodes_from(range(len(positions)))
        G.add_edges_from(pairs)
        components = list(nx.connected_components(G))
        halos = [comp for comp in components if len(comp) >= min_size]
        return halos

                targets = delta.ravel()[flat_idx].astype(np.float32).reshape(-1, 1)

                inputs_t = torch.from_numpy(inputs_np)
                targets_t = torch.from_numpy(targets)

                pred = model(inputs_t)
                loss = loss_fn(pred, targets_t)

                optimizer.step()
                epoch_loss += loss.item()

            # Physics-informed loss (linear growth rate enforcement)

                coords_col = torch.rand((n_col_phys, 3), dtype=torch.float32)

                (lambda_phys * phys_loss).backward()
                optimizer.step()
                epoch_loss += lambda_phys * phys_loss.item()

            scheduler.step()
            if epoch % 200 == 0:
                print(f"   Epoch {epoch:3d} | avg loss = {epoch_loss / len(delta_history):.2e} | lr={scheduler.get_last_lr()[0]:.2e}")

        print("✅ Real PINN surrogate trained (data-driven on full history)")
