        poly_term = (1 + 3.89 * q + (16.1 * q)**2 + (5.46 * q)**3 + (6.71 * q)**4)**-0.25
        return ln_term / (2.34 * q) * poly_term

    def compute_sigma8(self):

            return k**self.ns * self.transfer(k)**2
        def integrand(k):
            y = k * R
            W = 3 * (np.sin(y) - y * np.cos(y)) / y**3
            return k**2 / (2 * np.pi**2) * pk_func(k) * W**2
        sigma2 = quad(integrand, 1e-4, 10.0, epsrel=1e-4)[0]
        return np.sqrt(sigma2)

    @jit

    # ====================== MULTI-GRID POISSON (new in v9) ======================
    @jit
    def _relax_jacobi(self, u, rhs):
        """Jacobi relaxation for lap u = rhs (periodic)."""
        sum_neighbors = (
            jnp.roll(u, 1, axis=0) + jnp.roll(u, -1, axis=0) +
            jnp.roll(u, 1, axis=1) + jnp.roll(u, -1, axis=1) +
            jnp.roll(u, 1, axis=2) + jnp.roll(u, -1, axis=2)
        )
        u_new = (sum_neighbors + self.dx**2 * rhs) / 6.0
        return u_new

    @jit
    def _restrict(self, field):
        """Restriction to coarse grid by averaging."""
        c = (
            field[::2, ::2, ::2] + field[1::2, ::2, ::2] + field[::2, 1::2, ::2] + field[::2, ::2, 1::2] +
            field[1::2, 1::2, ::2] + field[1::2, ::2, 1::2] + field[::2, 1::2, 1::2] + field[1::2, 1::2, 1::2]
        ) / 8.0
        return c

    @jit
    def _prolong(self, coarse):
        """Prolongation to fine grid by linear interpolation."""
        shape_fine = [s * 2 for s in coarse.shape]
        fine = jnp.zeros(shape_fine, dtype=coarse.dtype)
        fine = fine.at[::2, ::2, ::2].set(coarse)

        # Interpolate along x
        fine = fine.at[1::2, ::, ::].set(0.5 * (fine[0::2, ::, ::] + fine[2::2, ::, ::]))

        # Interpolate along y
        fine = fine.at[:, 1::2, ::].set(0.5 * (fine[:, 0::2, ::] + fine[:, 2::2, ::]))

        # Interpolate along z
        fine = fine.at[:, ::, 1::2].set(0.5 * (fine[:, ::, 0::2] + fine[:, ::, 2::2]))

        return fine

    def _v_cycle(self, u, rhs, level, max_level):
        if level == 0:
            # Bottom level: use spectral solver
            return self._compute_potential_jax(rhs / (self.dx**2), 1.0)  # Adjust for lap u = rhs

        # Pre-relax
        for _ in range(5):
            u = self._relax_jacobi(u, rhs)

        # Residual
        sum_neighbors = (
            jnp.roll(u, 1, axis=0) + jnp.roll(u, -1, axis=0) +
            jnp.roll(u, 1, axis=1) + jnp.roll(u, -1, axis=1) +
            jnp.roll(u, 1, axis=2) + jnp.roll(u, -1, axis=2)
        )
        res = rhs - (sum_neighbors - 6 * u) / self.dx**2

        # Restrict
        res_coarse = self._restrict(res)

        # Recurse
        err_coarse = self._v_cycle(jnp.zeros_like(res_coarse), res_coarse, level - 1, max_level)

        # Prolong
        err = self._prolong(err_coarse)

        # Correct
        u += err

        # Post-relax
        for _ in range(5):
            u = self._relax_jacobi(u, rhs)

        return u

    @jit
    def _multi_grid_poisson_jax(self, delta, a):
        const = 1.5 * self.Omega_m / a
        rhs = const * delta  # lap phi = const * delta
        u = jnp.zeros_like(delta)
        max_level = int(jnp.log2(self.Ngrid)) - 2  # e.g., 6-2=4 for 64
        phi = self._v_cycle(u, rhs, max_level, max_level)
        return phi

    def _poisson_fft_3d(self, delta, a):
        """Public wrapper – uses multi-grid JAX Poisson solver."""
        delta_jax = jnp.asarray(delta, dtype=jnp.float32)
        phi = self._multi_grid_poisson_jax(delta_jax, float(a))
        return np.asarray(phi).astype(np.float64)

    # ====================== JAX CIC (new in v9, GPU-portable) ======================
    @jit
    def _deposit_cic_jax(self, pos):
        """JAX vectorized CIC deposition with scatter_add."""
        pos = jnp.asarray(pos)  # Assume (Np, 3)
        N = self.Ngrid
        dx = self.dx
        pos_scaled = pos / dx
        i0 = jnp.floor(pos_scaled).astype(jnp.int32) % N
        f = pos_scaled - i0
        one_minus_f = 1 - f
        offsets = jnp.array(list(itertools.product([0, 1], repeat=3)))  # (8, 3)
        Np = pos.shape[0]
        num_offsets = offsets.shape[0]

        # Compute weights for each offset
        w = jnp.prod(one_minus_f[:, None, :] ** (1 - offsets[None, :, :]) * f[:, None, :] ** offsets[None, :, :], axis=2)  # (Np, 8)

        # Compute indices for each offset
        idx = (i0[:, None, :] + offsets[None, :, :]) % N  # (Np, 8, 3)

        # Flatten
        w_flat = w.ravel()
        idx_flat = idx.reshape(-1, 3)

        # Flat indices
        flat_idx = idx_flat[:, 0] * N**2 + idx_flat[:, 1] * N + idx_flat[:, 2]

        # Scatter add
        rho_flat = lax.scatter_add(jnp.zeros(N**3, dtype=jnp.float32), flat_idx, w_flat, 
                                   lax.ScatterDimensionNumbers(update_window_dims=(), inserted_window_dims=(0,), scatter_dims_to_operand_dims=(0,)),
                                   indices_are_sorted=False, unique_indices=False, mode='add')

        rho = rho_flat.reshape((N, N, N))
        rho *= dx**3
        return rho

    @jit
    def _interpolate_cic_jax(self, field, pos):
        """JAX vmap-ed CIC interpolation."""
        def interpolate_single(p):
            i0 = jnp.floor(p / self.dx).astype(jnp.int32) % self.Ngrid
            f = p / self.dx - i0
            one_minus_f = 1 - f
            val = 0.0
            for dx_ in [0, 1]:
                for dy_ in [0, 1]:
                    for dz_ in [0, 1]:
                        w = (one_minus_f[0] if dx_ == 0 else f[0]) * \
                            (one_minus_f[1] if dy_ == 0 else f[1]) * \
                            (one_minus_f[2] if dz_ == 0 else f[2])
                        idx = (i0 + jnp.array([dx_, dy_, dz_])) % self.Ngrid
                        val += w * field[idx[0], idx[1], idx[2]]
            return val

        return vmap(interpolate_single)(pos)

    # ====================== LEAPFROG KDK INTEGRATOR (now JAX/GPU-portable) ======================
    def leapfrog_kdk(self, x, y, z, vx, vy, vz, delta, a_old, da):
        """Proper staggered KDK with Hubble drag centering + spectral forces, ported to JAX."""
        pos = jnp.stack([x, y, z], axis=-1)  # (Np, 3)
        vel = jnp.stack([vx, vy, vz], axis=-1)  # (Np, 3)
        delta = jnp.asarray(delta)

        a_new = a_old + da
        a_mid = (a_old + a_new) / 2.0
        H_mid = self.H(a_mid)
        dt = da / (a_mid * H_mid)

        # Kick ½ (old)
        fx, fy, fz = self._compute_forces_jax(delta, a_old)
        force_grid = jnp.stack([fx, fy, fz], axis=-1)  # (Ng, Ng, Ng, 3)
        force_p = self._interpolate_cic_jax(force_grid, pos)  # (Np, 3)
        vel = vel * (1.0 - (dt / 2) * self.H(a_old)) + (dt / 2) * force_p

        # Drift full step
        pos = (pos + vel * dt / a_mid) % self.Lbox

        # New delta
        rho = self._deposit_cic_jax(pos)
        new_delta = rho - 1.0

        # Kick ½ (new)
        fx, fy, fz = self._compute_forces_jax(new_delta, a_new)
        force_grid = jnp.stack([fx, fy, fz], axis=-1)
        force_p = self._interpolate_cic_jax(force_grid, pos)
        vel = vel * (1.0 - (dt / 2) * self.H(a_new)) + (dt / 2) * force_p

        x, y, z = pos[:, 0], pos[:, 1], pos[:, 2]
        vx, vy, vz = vel[:, 0], vel[:, 1], vel[:, 2]
        return np.asarray(x), np.asarray(y), np.asarray(z), np.asarray(vx), np.asarray(vy), np.asarray(vz), np.asarray(new_delta)

    # ====================== POWER SPECTRUM + LINEAR THEORY ======================
    def compute_power_spectrum(self, delta, nbins=40):
        delta_k = fftn(delta)
        k = np.sqrt(self.k2_mesh).flatten()
        pk = (np.abs(delta_k)**2 / self.Ngrid**6).flatten() * self.Lbox**3

        kbins = np.logspace(np.log10(0.01), np.log10(np.pi / self.dx), nbins)
        Pk = np.zeros(nbins-1)
        counts = np.zeros(nbins-1)
        for i in range(len(kbins)-1):
            mask = (k >= kbins[i]) & (k < kbins[i+1])
            if mask.sum() > 0:
                Pk[i] = pk[mask].mean()
                counts[i] = mask.sum()
        kcenters = (kbins[:-1] + kbins[1:]) / 2
        return kcenters[counts > 0], Pk[counts > 0]

    def linear_theory_pk(self, k):

    # ====================== 2LPT + HALO FINDER ======================
    def compute_2lpt(self, delta_lin, a_init):
        delta1 = delta_lin / self.growth_factor(a_init)
        prefactor = 1.5 * self.Omega_m / a_init
        phi1 = self._poisson_fft_3d(-delta1 / prefactor, a_init)

        k2 = self.k2_mesh
        phi1_k = fftn(phi1)
        dxx = np.real(ifftn(-self.kx**2 * phi1_k))
        dyy = np.real(ifftn(-self.ky**2 * phi1_k))
        dzz = np.real(ifftn(-self.kz**2 * phi1_k))
        dxy = np.real(ifftn(-self.kx * self.ky * phi1_k))
        dxz = np.real(ifftn(-self.kx * self.kz * phi1_k))
        dyz = np.real(ifftn(-self.ky * self.kz * phi1_k))

        source = dxx * dyy + dxx * dzz + dyy * dzz - dxy**2 - dxz**2 - dyz**2
        phi2 = self._poisson_fft_3d(source / prefactor, a_init)
        phi2_k = fftn(phi2)
        return phi1, phi2

    def find_halos(self, x, y, z, b=0.2, min_size=20):
        positions = np.stack([x.ravel(), y.ravel(), z.ravel()], axis=1)
        mean_sep = self.Lbox / self.Ngrid
        ll = b * mean_sep
        tree = KDTree(positions)
        pairs = tree.query_pairs(ll)
        G = nx.Graph()
        G.add_nodes_from(range(len(positions)))
        G.add_edges_from(pairs)
        components = list(nx.connected_components(G))
        halos = [comp for comp in components if len(comp) >= min_size]
        return halos

                d2phi_dz2 = compute_grad(dphi_dz, 2)
                lap_phi = d2phi_dx2 + d2phi_dy2 + d2phi_dz2
                const = torch.tensor(1.5 * self.Omega_m / a_col_np, dtype=torch.float32)
                r_poisson = lap_phi - const * delta_col

                # Continuity residual (linear): a H d_delta/da + div v = 0
                ddelta_da = compute_grad(delta_col, 3)
                div_v = compute_grad(v_col[:, 0], 0) + compute_grad(v_col[:, 1], 1) + compute_grad(v_col[:, 2], 2)
                H_a = torch.sqrt(self.Omega_m / a_col**3 + self.Omega_L)
                r_continuity = a_col.squeeze() * H_a * ddelta_da + div_v

                # Euler residual (linear): a H d_v/da + H v + (1/a) grad phi = 0
                dv_da = torch.stack([compute_grad(v_col[:, i], 3) for i in range(3)], dim=1)
                grad_phi = torch.stack([compute_grad(phi_col, i) for i in range(3)], dim=1)
                r_euler = a_col * H_a[:, None] * dv_da + H_a[:, None] * v_col + (1 / a_col[:, None]) * grad_phi

                phys_loss = loss_fn(r_poisson, torch.zeros_like(r_poisson)) + \
                            loss_fn(r_continuity, torch.zeros_like(r_continuity)) + \
                            loss_fn(r_euler, torch.zeros_like(r_euler))

                epoch_loss += lambda_phys * phys_loss.item()

                optimizer.zero_grad()
                (lambda_phys * phys_loss).backward()
                optimizer.step()

            scheduler.step()
            if epoch % 200 == 0:
                print(f"   Epoch {epoch:3d} | avg loss = {epoch_loss / len(delta_history):.2e} | lr={scheduler.get_last_lr()[0]:.2e}")

        self.pinn_model = model
        print("✅ Real PINN surrogate trained with full continuity + Poisson + Euler residuals")
        return model

    def predict_pinn(self, a):
        if not hasattr(self, 'pinn_model'):
            raise RuntimeError("Train PINN first")
        x_grid = np.linspace(0, 1, self.Ngrid)
        X, Y, Z = np.meshgrid(x_grid, x_grid, x_grid, indexing='ij')
        coords = np.stack([X, Y, Z], axis=-1).reshape(-1, 3)
        inputs = torch.tensor(np.hstack([coords, np.full((len(coords), 1), a)]), dtype=torch.float32)
        with torch.no_grad():
            outputs = self.pinn_model(inputs)
            delta = outputs[:, 0].numpy().reshape(self.Ngrid, self.Ngrid, self.Ngrid)
        return delta
