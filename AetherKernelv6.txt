@njit(fastmath=True, cache=True)
def _deposit_cic_numba(x, y, z, Lbox, Ngrid, dx):
    rho = np.zeros((Ngrid, Ngrid, Ngrid), dtype=np.float64)
    pos = np.stack((x, y, z), axis=-1) / dx
    i0 = np.floor(pos).astype(np.int32) % Ngrid
    f = pos - i0.astype(np.float64)
    one_minus_f = 1.0 - f
    Np = len(x)
    for dx_ in [0, 1]:
        for dy_ in [0, 1]:
            for dz_ in [0, 1]:
                w = (one_minus_f[:,0] if dx_ == 0 else f[:,0]) * \
                    (one_minus_f[:,1] if dy_ == 0 else f[:,1]) * \
                    (one_minus_f[:,2] if dz_ == 0 else f[:,2])
                idx0 = (i0[:,0] + dx_) % Ngrid
                idx1 = (i0[:,1] + dy_) % Ngrid
                idx2 = (i0[:,2] + dz_) % Ngrid
                for p in range(Np):
                    rho[idx0[p], idx1[p], idx2[p]] += w[p]
    rho *= dx ** 3
    return rho

@njit(fastmath=True, cache=True)
def _interpolate_cic_numba(field, x, y, z, Lbox, Ngrid, dx):
    val = np.zeros(len(x), dtype=np.float64)
    pos = np.stack((x, y, z), axis=-1) / dx
    i0 = np.floor(pos).astype(np.int32) % Ngrid
    f = pos - i0.astype(np.float64)
    one_minus_f = 1.0 - f
    Np = len(x)
    for dx_ in [0, 1]:
        for dy_ in [0, 1]:
            for dz_ in [0, 1]:
                w = (one_minus_f[:,0] if dx_ == 0 else f[:,0]) * \
                    (one_minus_f[:,1] if dy_ == 0 else f[:,1]) * \
                    (one_minus_f[:,2] if dz_ == 0 else f[:,2])
                idx0 = (i0[:,0] + dx_) % Ngrid
                idx1 = (i0[:,1] + dy_) % Ngrid
                idx2 = (i0[:,2] + dz_) % Ngrid
                for p in range(Np):
                    val[p] += w[p] * field[idx0[p], idx1[p], idx2[p]]
    return val

    def find_halos(self, x, y, z, b=0.2, min_size=20):

        Npart = len(positions)

        ll = b * mean_sep
        tree = KDTree(positions)

        G.add_edges_from(pairs)
        components = list(nx.connected_components(G))
        halos = [comp for comp in components if len(comp) >= min_size]
        return halos
